Class Trimesh

Trimesh.

example
// How to make a mesh with a single triangle
const vertices = [
    0, 0, 0, // vertex 0
    1, 0, 0, // vertex 1
    0, 1, 0  // vertex 2
]
const indices = [
    0, 1, 2  // triangle 0
]
const trimeshShape = new CANNON.Trimesh(vertices, indices)
